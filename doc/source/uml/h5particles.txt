@startuml
class H5Particles as "H5Particles(Group)" {
   _v_name : String = <name>
   particles : Group = ParticleItems
   bonds : Group = BondItems
   data : Table = Data
}

class ParticleItems as "ParticleItems(Group)" {
   _v_name : string = "particles"
   items : Table = ParticleTable
   data : Table = Data
}

class BondItems as "BondItems(Group)" {
   _v_name : string = "bonds"
   items : Table = BondsTable
   data : Table = Data
}

class ParticleTable as "ParticleTable(Table)" {
   _v_name : String = "Particles"
   row : ParticleTableRow
}

class BondsTable as "BondsTable(Table)" {
   _v_name : String = "Bonds"
   row : BondsTableRow
}

class Data as "Data(Table)" {
   _v_name: String = "Data"
   row : DataRow
}

class ParticleTableRow as "Particle(Row)" {
   [ uid ] : String[32]
   [ coordinates ] : Float64[3]
}

class BondTableRow as "Bond(Row)" {
   [ uid ] : String[32]
   [ particles ] : Integer
   [ n_particles ] : Integer = MAX_NUMBER_PARTICLES_IN_BOND
}

class DataRow as "Data(Row)" {
  [ uid ]: String[32]
  [ data ]: CUBARow
  [ mask ]: Bool[size of CUBARow]    
}

H5Particles -- ParticleItems
H5Particles -- BondItems
H5Particles -- Data

ParticleItems -- ParticleTable
BondItems -- BondsTable

ParticleTable -- ParticleTableRow
BondsTable -- BondTableRow
Data -- DataRow

@enduml
