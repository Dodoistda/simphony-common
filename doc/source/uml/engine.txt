  ..

  {abstract} get_mesh(name: str): ABCMesh
  {abstract} delete_mesh(name: str)
  {abstract} iter_meshes(names: iterable of str {optional}): iterator {unordered}
  ..
  {abstract} add_particles(particles: ABCParticles)
  {abstract} get_particles(name: str): ABCParticles
  {abstract} delete_particles(name: str)
  {abstract} iter_particles(names: iterable of str {optional}): iterator {unordered}
}
